/* https://leetcode.com/problems/managers-with-at-least-5-direct-reports */

with MngTab
as
(
    select managerId
    from Employee
    group by managerId
    having count ( id ) >= 5
)
select emp.name
from
    MngTab mng
        inner join
    Employee emp on mng.managerId = emp.id
